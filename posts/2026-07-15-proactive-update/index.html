н”„лЎңнҶ мҪң л Ҳмқҙм–ҙк°Җ sessionмқ„ кҙҖлҰ¬н•ҳм§Җ м•Ҡмқ„ лҝҗ, applicationмқҖ н•„мҡ”н•ҳлӢӨл©ҙ мһҗмӢ мқҳ лҚ°мқҙн„° лӘЁлҚё л Ҳмқҙм–ҙм—җм„ң stateлҘј кҙҖлҰ¬н•  мҲҳ мһҲлӢӨ. к·ёлҰ¬кі  MCP нҢҖмқҙ к¶ҢмһҘн•ҳлҠ” л°©мӢқмқҙ л°”лЎң Explicit Handle PatternмқҙлӢӨ.

4.1 Explicit Handle Patternмқҳ мӣҗлҰ¬

ServerлҠ” tool callмқҳ кІ°кіјлЎң opaque handle(basket_id, workflow_run_id, session_token л“ұ)мқ„ л°ңкёүн•ҳкі , model(client)мқҖ мқҙ handleмқ„ мқҙнӣ„ tool callмқҳ argumentлЎң лӢӨмӢң м „лӢ¬н•ңлӢӨ.

// === TypeScript: ExplicitHandleManager ===
interface SessionState {
  basketId: string;
  items: BasketItem[];
  createdAt: Date;
  expiresAt: Date;
}

class ExplicitHandleManager {
  private store: Map<string, SessionState> = new Map();

  /**
   * tool call кІ°кіјлЎң stateful handle л°ңкёү
   * - handle мһҗмІҙлҠ” opaque token
   * - serverл§Ңмқҙ handle вҶ’ state л§Өн•‘мқ„ м•Ңкі  мһҲмқҢ
   */
  createBasket(toolName: string, args: Record<string, unknown>): {
    resultType: "text";
    text: string;
    _handle: string;
  } {
    const basket: SessionState = {
      basketId: crypto.randomUUID(),
      items: [],
      createdAt: new Date(),
      expiresAt: new Date(Date.now() + 30 * 60 * 1000), // 30л¶„ TTL
    };
    this.store.set(basket.basketId, basket);

    return {
      resultType: "text",
      text: `Basket created: ${basket.basketId}`,
      // Handleмқ„ кІ°кіјмқҳ мқјл¶ҖлЎң л°ҳнҷҳ
      _handle: basket.basketId,
    };
  }

  /**
   * modelмқҙ м „лӢ¬н•ң handleлЎң stateful мһ‘м—… мҲҳн–ү
   * - modelмқҖ handleмқҳ мқҳлҜёлҘј м•Ң н•„мҡ” м—ҶмқҢ
   * - serverк°Җ handle вҶ’ state л§Өн•‘мқ„ мҲҳн–ү
   */
  addToBasket(basketId: string, item: string): {
    resultType: "text";
    text: string;
  } {
    const basket = this.store.get(basketId);

    if (!basket || basket.expiresAt < new Date()) {
      return {
        resultType: "text",
        text: "Error: Basket not found or expired. Please create a new basket.",
      };
    }

    basket.items.push({ id: crypto.randomUUID(), name: item, addedAt: new Date() });
    return {
      resultType: "text",
      text: `Added "${item}" to basket ${basketId}. Total items: ${basket.items.length}`,
    };
  }
}

4.2 Protocol-Level Session vs Explicit Handle: мҷң Explicit Handleмқҙ лҚ” к°•л Ҙн•ңк°Җ

MCP нҢҖмқҳ кіөмӢқ л¬ём„ңлҠ” мқҙл ҮкІҢ л§җн•ңлӢӨ:

"The model can compose handles across tools, reason about them, and hand them off between steps in ways that externally managed session state, hidden in transport metadata, never really allowed."

мҰү, protocol-level sessionмқҖ transport metadataлЎң к°җм¶°м ё мһҲм–ҙ modelмқҙ sessionмқҳ мЎҙмһ¬лҘј мқём§Җн•ҳм§Җ лӘ»н–ҲлӢӨ. л°ҳл©ҙ explicit handleмқҖ:

  1. Modelмқҙ handleмқ„ м§Ғм ‘ ліҙкі  м¶”лЎ н•  мҲҳ мһҲлӢӨ: "мқҙ basket_idлҘј к°Җм§„ мһҘл°”кө¬лӢҲм—җ л¬јкұҙмқ„ 3к°ң л„Јм—ҲмңјлӢҲ, мқҙм ң кІ°м ң toolмқ„ нҳём¶ңн•ҳмһҗ"
  2. Handleмқ„ cross-toolлЎң м „лӢ¬н•  мҲҳ мһҲлӢӨ: search toolмқҙ л°ҳнҷҳн•ң result_idлҘј, summarize toolм—җ м „лӢ¬
  3. Handleмқ„ cross-sessionмңјлЎң м „лӢ¬н•  мҲҳ мһҲлӢӨ: н•ң м„ём…ҳм—җм„ң мғқм„ұлҗң handleмқ„ лӢӨлҘё м„ём…ҳмқҳ modelм—җкІҢ м „лӢ¬
  4. Handleмқ„ лЎңк№…н•ҳкі  м¶”м Ғн•  мҲҳ мһҲлӢӨ: observability systemмқҙ handleмқ„ keyлЎң session state ліҖнҷ”лҘј м¶”м Ғ
// === Modelмқҳ Handle Reasoning мҳҲмӢң ===

interface InputRequiredResult {
  resultType: "inputRequired";
  inputRequests: Record<string, InputRequest>;
  requestState: string; // opaque token, serverк°Җ resumeм—җ н•„мҡ”н•ң лӘЁл“  м •ліҙлҘј мқёмҪ”л”©

interface InputRequest {
  type: "elicitation";
  message: string;
  schema: JSONSchema;
}

class ElicitationClient {
  /**
   * Serverк°Җ inputRequiredлҘј л°ҳнҷҳн•ҳл©ҙ,
   * clientк°Җ мһ…л Ҙмқ„ лӘЁм•„м„ң лӢӨмӢң к°ҷмқҖ мҡ”мІӯмқ„ re-issue
   */
  async handleElicitation(
    originalRequest: JSONRPCRequest,
    inputRequired: InputRequiredResult
  ): Promise<JSONRPCResponse> {
    // Step 1: inputRequestsлҘј мӮ¬мҡ©мһҗ/agentм—җкІҢ н‘ңмӢң
    const inputResponses: Record<string, unknown> = {};

    for (const [key, request] of Object.entries(inputRequired.inputRequests)) {
      if (request.type === "elicitation") {
        // User лҳҗлҠ” upstream agentм—җкІҢ мһ…л Ҙ мҡ”мІӯ
        inputResponses[key] = await this.promptUser(request.message, request.schema);
      }
    }

    // Step 2: к°ҷмқҖ methodм—җ inputResponses + requestStateлҘј мӢӨм–ҙ мһ¬м „мҶЎ
    const retryRequest: JSONRPCRequest = {
      ...originalRequest,
      params: {
        ...originalRequest.params,
        inputResponses,
        requestState: inputRequired.requestState,
      },
    };

    // Step 3: м–ҙл–Ө server instanceл“  мІҳлҰ¬ к°ҖлҠҘ
    // (requestStateм—җ лӘЁл“  resume м •ліҙк°Җ л“Өм–ҙмһҲмқҢ)
    return await this.transport.send(retryRequest);
  }

  private async promptUser(message: string, schema: JSONSchema): Promise<unknown> {
    // мӢӨм ң кө¬нҳ„: мӮ¬мҡ©мһҗм—җкІҢ dialog н‘ңмӢң лҳҗлҠ” upstream agent нҳём¶ң
    console.log(`[Elicitation Required] ${message}`);
    return schema.type === "boolean" ? true : prompt(message);
  }
}

// === TypeScript: Trace Context Propagation Middleware ===
class TracePropagationMiddleware {
  private tracer: opentelemetry.Tracer;

  /**
   * MCP requestмқҳ _metaм—җ trace contextлҘј мЈјмһ…
   * Host application вҶ’ Client SDK вҶ’ MCP Server вҶ’ Downstream API
   * м „мІҙк°Җ н•ҳлӮҳмқҳ span treeлЎң м—°кІ°лҗЁ
   */
  injectTraceContext(params: Record<string, unknown>): Record<string, unknown> {
    const span = opentelemetry.trace.getActiveSpan();
    if (!span) return params;

    const traceContext = span.spanContext();

    return {
      ...params,
      _meta: {
        ...(params._meta as Record<string, unknown>),
        // W3C traceparent: version-traceId-spanId-traceFlags
        traceparent: `00-${traceContext.traceId}-${traceContext.spanId}-${traceContext.traceFlags}`,
        // tracestate: vendor-specific trace data
        tracestate: `mcp=tool_call,otel=prod`,
        // baggage: propagation key-value pairs
        baggage: `user_id=sjlee,request_id=req_abc123`,
      },
    };
  }
}

"м–ҙл–»кІҢ handleм—җ лӢҙкёҙ dataк°Җ ліҖмЎ°лҗҳм§Җ м•Ҡм•ҳлҠ”м§Җ ліҙмһҘн•  кІғмқёк°Җ?" вҖ” protocol layerмқҳ л¬ём ңк°Җ мӮ¬лқјм§„ лҢҖмӢ , data model layerм—җм„ң лҸҷмқјн•ң л¬ём ңлҘј н•ҙкІ°н•ҙм•ј н•ңлӢӨ.

8.2 Stateless Protocol + Explicit Handle + ZK Proof

к°ҖмһҘ нқҘлҜёлЎңмҡҙ мЎ°н•©мқҖ Stateless Protocol + Explicit Handle + Zero-Knowledge ProofлӢӨ.

// === Cross-Trust Verification Handle ===
// Server Aк°Җ л°ңкёүн•ң handleмқ„ Server Bк°Җ кІҖмҰқ
interface VerifiableHandle {
  // opaque handle (Server Aмқҳ лӮҙл¶Җ session key)
  handle: string;

  // Zero-knowledge proof of handle validity
  // Server BлҠ” мқҙ proofлЎң handleмқҙ мң„мЎ°лҗҳм§Җ м•Ҡм•ҳмқҢмқ„ кІҖмҰқ
  // Server BлҠ” handleмқҳ лӮҙл¶Җ лҚ°мқҙн„°лҘј м•Ң н•„мҡ” м—ҶмқҢ
  proof: {
    type: "zkp-snark" | "signed-jwt" | "merkle-proof";
    value: string;

    verifier: string;

  expiresAt: string;
}

9. мһҗк°Җ кІҖнҶ  (Self-Critique)

мқҙ кёҖмқҳ н•ңкі„мҷҖ м°Ёлі„нҷ” м§Җм җмқ„ лӘ…мӢңн•ңлӢӨ:

  1. #055 Context Engineeringкіјмқҳ м°Ёлі„нҷ”: #055лҠ” MCP session лӮҙм—җм„ң context windowлҘј кҙҖлҰ¬н•ҳлҠ” application-level м „лһөмқ„ лӢӨлӨҳлӢӨ. ліё кёҖмқҖ session мһҗмІҙк°Җ protocol layerм—җм„ң м ңкұ°лҗң мғҲлЎңмҡҙ spec мһҗмІҙлҘј л¶„м„қн•ңлӢӨ. #055лҠ” "sessionмқҙ мһҲлӢӨлҠ” м „м ң н•ҳм—җ м–ҙл–»кІҢ мһҳ м“ёк№Ң"мҳҖлӢӨл©ҙ, ліё кёҖмқҖ "sessionмқҙ м—Ҷм–ҙмЎҢлҠ”лҚ° м–ҙл–»кІҢ лҢҖмІҳн• к№Ң"лӢӨ.

  2. #056 Context ObservabilityмҷҖмқҳ м°Ёлі„нҷ”: #056мқҖ session evictionмқ„ м¶”м Ғн•ҳлҠ” observability layer кө¬м¶•мқҙ н•өмӢ¬мқҙм—ҲлӢӨ. ліё кёҖмқҖ protocol layerмқҳ statelessнҷ”к°Җ observabilityм—җ W3C Trace Context кё°л°ҳмқҳ н‘ңмӨҖм Ғмқё м ‘к·јмқ„ м ңкіөн•ҳлҠ” м җмқ„ лӢӨлЈ¬лӢӨ. лӢЁ, ліё кёҖмқҖ observabilityліҙлӢӨлҠ” protocol architectureм—җ мҙҲм җмқ„ л§һм¶”л©°, tracingмқҖ к·ё мқјл¶ҖлЎңл§Ң лӢӨлЈ¬лӢӨ.

  3. #059 Cross-Trust HandoffмҷҖмқҳ м°Ёлі„нҷ”: #059лҠ” лӢӨлҘё trust domain к°„ session м „лӢ¬мқ„ zero-knowledge proofлЎң н•ҙкІ°н•ҳлҠ” л°©лІ•мқҙм—ҲлӢӨ. ліё кёҖмқҖ protocol session мһҗмІҙмқҳ м ңкұ°к°Җ cross-trust scenarioм—җ м–ҙл–Ө мҳҒн–Ҙмқ„ мЈјлҠ”м§Җ л¶„м„қн•ңлӢӨ. мӨ‘мҡ”н•ң кІғмқҖ: protocol session м ңкұ°лҠ” cross-trust л¬ём ңлҘј н•ҙкІ°н•ҳм§Җ м•Ҡкі , л¬ём ңмқҳ м„ұкІ©мқ„ protocol layerм—җм„ң data model layerлЎң мқҙлҸҷмӢңнӮЁлӢӨлҠ” м җмқҙлӢӨ.

  4. мқҙ кёҖмқҳ н•ңкі„:

    • 2026-07-28 RCлҠ” м•„м§Ғ мөңмў… specмқҙ м•„лӢҲлӢӨ(7мӣ” 28мқј нҷ•м •). мқјл¶Җ м„ёл¶ҖмӮ¬н•ӯмқҙ ліҖкІҪлҗ  мҲҳ мһҲлӢӨ.
    • Explicit Handle PatternмқҖ MCP нҢҖмқҙ к¶ҢмһҘн•ҳлҠ” нҢЁн„ҙмқҙм§Җл§Ң, мқҙкІғмқҙ мң мқјн•ң л°©лІ•мқҖ м•„лӢҲлӢӨ. OAuth 2.1 session token, signed JWT claim л“ұ лӢӨм–‘н•ң м ‘к·јмқҙ к°ҖлҠҘн•ҳлӢӨ.
    • Tasks Extensionмқҳ кө¬мІҙм Ғмқё кө¬нҳ„кіј м„ұлҠҘмқҖ SDKк°Җ м•Ҳм •нҷ”лҗң нӣ„ нҸүк°Җн•ҙм•ј н•ңлӢӨ.
    • OAuth 2.1 + PKCE мқҳл¬ҙнҷ”лҠ” remote serverм—җ н•ңм •лҗҳл©°, local(stdio) transportлҠ” н•ҙлӢ№н•ҳм§Җ м•ҠлҠ”лӢӨ.

  5. лӢӨмқҢ кёҖмқҳ л°©н–Ҙм„ұ: Sessionмқҙ data model layerлЎң мқҙлҸҷн•ҳл©ҙм„ң, "м—җмқҙм „нҠёк°Җ cross-trust мӢңлӮҳлҰ¬мҳӨм—җм„ң data model levelмқҳ session мғҒнғңлҘј м–ҙл–»кІҢ м•Ҳм „н•ҳкІҢ мЈјкі л°ӣмқ„ кІғмқёк°Җ"к°Җ лӢӨмқҢ кіјм ңк°Җ лҗңлӢӨ. мқҙлҠ” Data Model Levelмқҳ Session State Synchronization ProtocolмқҙлқјлҠ” мғҲлЎңмҡҙ мЈјм ңлЎң мқҙм–ҙм§Ҳ мҲҳ мһҲлӢӨ. мҰү, protocol layerмқҳ лӢЁмҲңнҷ”лҠ” data model layerмқҳ ліөмһЎм„ұмқ„ мҰқк°ҖмӢңмј°кі , мқҙ ліөмһЎм„ұмқ„ н•ҙкІ°н•ҳкё° мң„н•ң н‘ңмӨҖнҷ”лҗң м ‘к·јмқҙ н•„мҡ”н•ҳлӢӨ.

10. кІ°лЎ 

MCP 2026-07-28мқҳ stateless м „нҷҳмқҖ лӢЁмҲңн•ң Transport м—…к·ёл Ҳмқҙл“ңк°Җ м•„лӢҲлқј "м—°кІ° мӨ‘мӢ¬(connection-oriented)"м—җм„ң "л©”мӢңм§Җ мӨ‘мӢ¬(message-oriented)"мңјлЎңмқҳ м„Өкі„ мІ н•ҷ м „нҷҳмқҙлӢӨ.

ліҖкІҪмқҳ н•өмӢ¬ л©”мӢңм§ҖлҠ” лӢӨмқҢкіј к°ҷлӢӨ:

  1. н”„лЎңнҶ мҪңмқҖ лӢЁмҲңн•ҙм•ј н•ңлӢӨ: м—°кІ° кҙҖлҰ¬, session state, transport metadataлҘј protocol layerк°Җ кҙҖлҰ¬н•ҳлҠ” кІғмқҖ ліөмһЎн•ҳлӢӨ. ProtocolмқҖ message routingкіј formatл§Ң м •мқҳн•ҳкі , лӮҳлЁём§ҖлҠ” мғҒмң„ layerм—җ мң„мһ„н•ҳмһҗ.

  2. Modelм—җкІҢ л…ём¶ңн•ҳлқј: Session IDлҘј transport metadataлЎң мҲЁкё°лҠ” лҢҖмӢ , explicit handleмқ„ tool argumentлЎң л…ём¶ңн•ҳл©ҙ modelмқҙ stateлҘј м¶”лЎ н•ҳкі  мЎ°мһ‘н•  мҲҳ мһҲлӢӨ.

  3. мқён”„лқјмҷҖ нҳёнҷҳлҗҳм–ҙм•ј н•ңлӢӨ: sticky session, shared session store, deep packet inspectionмқҖ н‘ңмӨҖ HTTP мқён”„лқјмҷҖ нҳёнҷҳлҗҳм§Җ м•ҠлҠ”лӢӨ. Round-robin LB, standard caching, header-based routing вҖ” мқҙлҜё кІҖмҰқлҗң мқён”„лқј мң„м—җм„ң лҸҷмһ‘н•ҳкІҢ н•ҳмһҗ.

  4. ліҙм•ҲмқҖ protocolмқҙ м•„лӢҲлқј data layerм—җм„ң: OAuth 2.1, PKCE, signed JWT, ZK proof вҖ” protocolмқҙ м•„лӢҲлқј data layerмқҳ м•”нҳён•ҷм Ғ л©”м»ӨлӢҲмҰҳмқҙ ліҙм•Ҳмқ„ лӢҙлӢ№н•ҳкІҢ н•ҳмһҗ.

2026л…„ 7мӣ” 28мқј, MCPлҠ” мӢӨн—ҳм Ғмқё н”„лЎңнҶ мҪңм—җм„ң н”„лЎңлҚ•м…ҳ мқён”„лқјлЎң лҸ„м•Ҫн•ңлӢӨ. к·ёлҰ¬кі  мқҙ ліҖнҷ”лҠ” Agent мғқнғңкі„ м „мІҙмқҳ м•„нӮӨн…ҚмІҳ м„Өкі„м—җ мғҲлЎңмҡҙ кё°мӨҖмқ„ м ңмӢңн•  кІғмқҙлӢӨ.
