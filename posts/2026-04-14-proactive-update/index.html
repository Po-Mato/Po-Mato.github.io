from dataclasses import dataclass, field
from datetime import datetime
from typing import TypedDict, Protocol, Optional
from enum import Enum
import json


class MemoryTier(Enum):
    SHORT_TERM = "short_term"    # нҳ„мһ¬ turn / attention window
    SESSION = "session"          # нҳ„мһ¬ м„ём…ҳ м „мІҙ
    MEDIUM_TERM = "medium_term"  # мөңк·ј Nмқј нҷңлҸҷ мҡ”м•Ҫ
    LONG_TERM = "long_term"      # мҳҒкө¬ м ҖмһҘмҶҢ (vector/KG)


@dataclass
class MemoryEntry:
    """л©”лӘЁлҰ¬ н•ӯлӘ© н•ҳлӮҳ"""
    id: str
    content: str
    tier: MemoryTier
    created_at: datetime
    accessed_at: datetime
    access_count: int = 0
    importance: float = 1.0       # 0.0 ~ 1.0 вҖ” forget кІ°м •м—җ мӮ¬мҡ©
    embedding: list[float] | None = None
    metadata: dict = field(default_factory=dict)

    def touch(self):
        """м ‘к·ј мӢң access_count мҰқк°Җ л°Ҹ accessed_at к°ұмӢ """
        self.access_count += 1
        self.accessed_at = datetime.now()


class AgentMemorySystem:
    """
    3кі„мёө л©”лӘЁлҰ¬ мӢңмҠӨн…ңмқҳ н•өмӢ¬ мқён„°нҺҳмқҙмҠӨ.
    к°Ғ tierлі„лЎң лӢӨлҘё м ҖмһҘмҶҢмҷҖ TTL м •мұ…мқҙ м Ғмҡ©лҗңлӢӨ.
    """

    def __init__(
        self,
        short_term_limit: int = 10,      # STM: мөңк·ј 10к°ң л©”мӢңм§Җ
        session_limit: int = 200,         # м„ём…ҳ: мөңк·ј 200к°ң л©”мӢңм§Җ
        medium_term_days: int = 7,        # мӨ‘кё°: 7мқјм№ҳ rolling мҡ”м•Ҫ
    ):
        self.short_term_limit = short_term_limit
        self.session_limit = session_limit
        self.medium_term_days = medium_term_days

        # к°Ғ tierлі„ м ҖмһҘмҶҢ
        self.short_term: list[MemoryEntry] = []   # лҰ¬мҠӨнҠё (in-memory)
        self.session: list[MemoryEntry] = []       # лҰ¬мҠӨнҠё (in-memory, larger)
        self.medium_term: list[MemoryEntry] = []  # Rolling мҡ”м•Ҫ buffer
        self.long_term: VectorStore | None = None  # Vector store (Pinecone, Qdrant л“ұ)

    def add(self, content: str, tier: MemoryTier, metadata: dict = None) -> MemoryEntry:
        """л©”лӘЁлҰ¬м—җ н•ӯлӘ© м¶”к°Җ"""
        entry = MemoryEntry(
            id=f"{tier.value}_{datetime.now().isoformat()}",
            content=content,
            tier=tier,
            created_at=datetime.now(),
            accessed_at=datetime.now(),
            metadata=metadata or {},
        )
        self._store(entry)
        return entry

    def retrieve(self, query: str, tier: MemoryTier, top_k: int = 5) -> list[MemoryEntry]:
        """л©”лӘЁлҰ¬м—җм„ң кІҖмғү"""
        if tier == MemoryTier.LONG_TERM and self.long_term:
            return self.long_term.search(query, top_k)
        elif tier == MemoryTier.SESSION:
            # лӢЁмҲң мөңк·јм ‘ кІҖмғү (BM25 лҳҗлҠ” embedding)
            return self._session_search(query, top_k)
        elif tier == MemoryTier.MEDIUM_TERM:
            return self._rolling_search(query, top_k)
        return []

    def build_context(self, query: str) -> str:
        """LLMм—җ м „лӢ¬н•  нҶөн•© м»Ён…ҚмҠӨнҠё л¬ёмһҗм—ҙ кө¬м„ұ"""
        parts = []

        # 1мҲңмң„: STM вҖ” нҳ„мһ¬ мһ‘м—… кҙҖл Ё мөңк·ј н•ӯлӘ©
        stm_results = self.short_term[-self.short_term_limit:]
        if stm_results:
            parts.append(f"[лӢЁкё° кё°м–ө вҖ” мөңк·ј {len(stm_results)}к°ң]\n" +
                          "\n".join(f"- {e.content}" for e in stm_results))

        # 2мҲңмң„: м„ём…ҳ вҖ” кҙҖл Ё кё°м–ө кІҖмғү
        session_results = self.retrieve(query, MemoryTier.SESSION, top_k=5)
        if session_results:
            parts.append(f"[м„ём…ҳ кё°м–ө]\n" +
                          "\n".join(f"- {e.content}" for e in session_results))

        # 3мҲңмң„: мһҘкё° вҖ” лІЎн„° кІҖмғү
        long_term_results = self.retrieve(query, MemoryTier.LONG_TERM, top_k=3)
        if long_term_results:
            parts.append(f"[мһҘкё° кё°м–ө]\n" +
                          "\n".join(f"- {e.content}" for e in long_term_results))

        return "\n\n".join(parts) if parts else "(кё°м–ө м—ҶмқҢ)"

    def _store(self, entry: MemoryEntry):
        if entry.tier == MemoryTier.SHORT_TERM:
            self.short_term.append(entry)
            self._enforce_tier_limit(MemoryTier.SHORT_TERM)
        elif entry.tier == MemoryTier.SESSION:
            self.session.append(entry)
            self._enforce_tier_limit(MemoryTier.SESSION)
        elif entry.tier == MemoryTier.MEDIUM_TERM:
            self.medium_term.append(entry)
        elif entry.tier == MemoryTier.LONG_TERM:
            if self.long_term:
                self.long_term.upsert(entry)

    def _enforce_tier_limit(self, tier: MemoryTier):
        if tier == MemoryTier.SHORT_TERM:
            while len(self.short_term) > self.short_term_limit:
                self.short_term.pop(0)
        elif tier == MemoryTier.SESSION:
            while len(self.session) > self.session_limit:
                self.session.pop(0)

STM кҙҖлҰ¬ м „лһө: Importance-Weighted Eviction

лӢЁмҲңнһҲ "мөңк·ј Nк°ң"л§Ң мң м§Җн•ҳлҠ” FIFO л°©мӢқмқҖ мӨ‘мҡ”н•ң м •ліҙк°Җ мЎ°кё°м—җ м ңкұ°лҗ  мҲҳ мһҲлӢӨ. Importance-weighted evictionмқҖ м ‘к·ј л№ҲлҸ„мҷҖ мӨ‘мҡ”лҸ„лҘј н•Ёк»ҳ кі л Өн•ңлӢӨ:

from collections import deque
import heapq

class ImportanceWeightedSTM:
    """
    Importance м җмҲҳ кё°л°ҳ лӢЁкё° кё°м–ө кҙҖлҰ¬.
    - мөңк·ј м ‘к·ј: access_count к°ҖмӨ‘
    - мӨ‘мҡ”лҸ„: metadataм—җм„ң м§Җм • (user Explicitly marked)
    - л№„мҡ©: нҶ нҒ° кёёмқҙм—җ л№„лЎҖн•ҳм—¬ мң м§Җ л№„мҡ© мҰқк°Җ
    """

    def __init__(self, capacity: int = 10, max_tokens: int = 3000):
        self.capacity = capacity
        self.max_tokens = max_tokens
        self.entries: list[MemoryEntry] = []
        self.current_tokens: int = 0

    def _score(self, entry: MemoryEntry) -> float:
        """Importance scoring function"""
        recency_weight = 1.0 / (1.0 + (datetime.now() - entry.accessed_at).seconds / 3600)
        access_weight = min(entry.access_count / 10.0, 1.0)
        importance_weight = entry.importance
        token_penalty = 1.0 / (1.0 + len(entry.content) / 500)
        return (recency_weight * 0.3 + access_weight * 0.3 +
                importance_weight * 0.3 + token_penalty * 0.1)

    def add(self, entry: MemoryEntry) -> bool:
        """н•ӯлӘ© м¶”к°Җ вҖ” мҡ©лҹү мҙҲкіј мӢң eviction нӣ„ м¶”к°Җ"""
        entry_tokens = len(entry.content) // 4  # лҢҖлһөм Ғ нҶ нҒ° мҲҳ

        # мҡ©лҹү лҳҗлҠ” нҶ нҒ° мҙҲкіј мӢң eviction
        while (len(self.entries) >= self.capacity or
               self.current_tokens + entry_tokens > self.max_tokens):
            if not self.entries:
                break
            # к°ҖмһҘ лӮ®мқҖ м җмҲҳмқҳ н•ӯлӘ© м ңкұ°
            evict_idx = min(
                range(len(self.entries)),
                key=lambda i: self._score(self.entries[i])
            )
            evicted = self.entries.pop(evict_idx)
            self.current_tokens -= len(evicted.content) // 4

        if entry_tokens <= self.max_tokens:
            entry.touch()
            self.entries.append(entry)
            self.current_tokens += entry_tokens
            return True
        return False

    - Level 0: мӣҗліё л©”мӢңм§Җ (мҷ„м „н•ң лҢҖнҷ” кё°лЎқ)
    - Level 1: Turn лӢЁмң„ мҡ”м•Ҫ (к°Ғ л©”мӢңм§Җ мҢҚмқ„ н•ҳлӮҳмқҳ л¬ёлӢЁмңјлЎң)
    - Level 2: мЈјм ң лӢЁмң„ мҡ”м•Ҫ (л°ҳліөлҗҳлҠ” мЈјм ңлі„лЎң кІ°н•©)
    - Level 3: м„ём…ҳ м „мІҙ мҡ”м•Ҫ (2~3 л¬ёмһҘ)
    """

    def __init__(self, llm, max_turns_per_summary: int = 5):
        self.llm = llm
        self.max_turns_per_summary = max_turns_per_summary
        self.levels: dict[int, list[MemoryEntry]] = {
            0: [],   # мӣҗліё
            1: [],   # turn мҡ”м•Ҫ
            2: [],   # topic мҡ”м•Ҫ
            3: [],   # session мҡ”м•Ҫ
        }

    async def add_message(self, role: str, content: str, metadata: dict = None):
        """л©”мӢңм§Җ м¶”к°Җ вҖ” мҡ”м•Ҫ мЎ°кұҙ м¶©мЎұ мӢң мғҒмң„ кі„мёө мғқм„ұ"""
        entry = MemoryEntry(
            id=f"msg_{len(self.levels[0]) + 1}",
            content=f"[{role}]: {content}",
            tier=MemoryTier.SESSION,
            created_at=datetime.now(),
            accessed_at=datetime.now(),
            metadata={**(metadata or {}), "role": role},
        )
        self.levels[0].append(entry)

        # Level 0мқҙ Nк°ң лӘЁмқҙл©ҙ Level 1 мҡ”м•Ҫ мғқм„ұ
        if len(self.levels[0]) % self.max_turns_per_summary == 0:
            await self._summarize_to_level(0, 1)

        # Level 1мқҙ Nк°ң лӘЁмқҙл©ҙ Level 2 мғқм„ұ
        if len(self.levels[1]) >= 4:
            await self._summarize_to_level(1, 2)

    async def _summarize_to_level(self, from_level: int, to_level: int):
        """from_levelмқҳ н•ӯлӘ©л“Өмқ„ to_levelмңјлЎң мҡ”м•Ҫ"""
        source_content = "\n".join(e.content for e in self.levels[from_level][-self.max_turns_per_summary:])

        prompt = f"""лӢӨмқҢ лҢҖнҷ”лҘј {to_level}лӢЁкі„ мҡ”м•ҪмңјлЎң м••м¶•н•ҙмӨҳ:
- л„Ҳл¬ҙ мһҘнҷ©н•ҳм§Җ м•ҠкІҢ
- н•өмӢ¬ м •ліҙмҷҖ кІ°м • мӮ¬н•ӯмқҖ л°ҳл“ңмӢң нҸ¬н•Ё
- {to_level} л ҲлІЁмқјмҲҳлЎқ лҚ” м••м¶•м ҒмңјлЎң

---
{source_content}
---
"""
        summary_text = await self.llm.agenerate([prompt])
        summary_entry = MemoryEntry(
            id=f"summary_l{to_level}_{datetime.now().isoformat()}",
            content=summary_text,
            tier=MemoryTier.SESSION,
            created_at=datetime.now(),
            accessed_at=datetime.now(),
            importance=0.8,
            metadata={"level": to_level, "source_count": self.max_turns_per_summary},
        )
        self.levels[to_level].append(summary_entry)

    def get_context(self, query: str) -> str:
        """м„ём…ҳ кё°м–өм—җм„ң мҝјлҰ¬м—җ кҙҖл Ёлҗң м»Ён…ҚмҠӨнҠё л°ҳнҷҳ"""
        context_parts = []

        # Session мҡ”м•Ҫ (Level 3) вҖ” н•ӯмғҒ лЁјм Җ
        if self.levels[3]:
            context_parts.append(f"[м„ём…ҳ мҡ”м•Ҫ] {self.levels[3][-1].content}")

        # Topic мҡ”м•Ҫ (Level 2) вҖ” мҝјлҰ¬ кҙҖл Ём„ұ н•„н„°
        topic_relevant = [e for e in self.levels[2] if any(w in e.content for w in query.split()[:3])]
        if topic_relevant:
            context_parts.append("[мЈјм ң мҡ”м•Ҫ]\n" + "\n".join(e.content for e in topic_relevant[-3:]))

        # Turn мҡ”м•Ҫ (Level 1) вҖ” мөңк·ј кІғл§Ң
        if self.levels[1]:
            context_parts.append("[мөңк·ј лҢҖнҷ” нқҗлҰ„]\n" + "\n".join(e.content for e in self.levels[1][-3:]))

        return "\n\n".join(context_parts) if context_parts else ""

from dataclasses import dataclass
from typing import Protocol
import numpy as np


class VectorStore(Protocol):
    async def upsert(self, entry: MemoryEntry): ...
    async def search(self, query: str, top_k: int) -> list[MemoryEntry]: ...


class HybridLongTermMemory:
    """
    Vector search + Keyword search (BM25) + Metadata filterлҘј
    мЎ°н•©н•ң н•ҳмқҙлёҢлҰ¬л“ң мһҘкё° кё°м–ө.
    """

    def __init__(
        self,
        vector_store: VectorStore,
        embedder,          # OpenAI embeddings лҳҗлҠ” sentence-transformers
        alpha: float = 0.7,  # vector weight (1-alpha = keyword weight)
    ):
        self.vector_store = vector_store
        self.embedder = embedder
        self.alpha = alpha
        self.keyword_index: dict[str, list[str]] = {}  # term -> entry_ids

    async def add(self, content: str, metadata: dict = None) -> MemoryEntry:
        """мһҘкё° кё°м–өм—җ м ҖмһҘ вҖ” vector мһ„лІ л”©кіј keyword мқёлҚұмӢұ лҸҷмӢң мҲҳн–ү"""
        entry = MemoryEntry(
            id=f"ltm_{datetime.now().isoformat()}",
            content=content,
            tier=MemoryTier.LONG_TERM,
            created_at=datetime.now(),
            accessed_at=datetime.now(),
            metadata=metadata or {},
        )

        # Vector embedding
        entry.embedding = await self.embedder.embed(content)

        # Keyword index (к°„лӢЁн•ң BM25 лҢҖмӢ  л№ҲлҸ„ кё°л°ҳ)
        words = content.lower().split()
        for word in set(words):
            if len(word) > 3:  # stop word м ңкұ°
                if word not in self.keyword_index:
                    self.keyword_index[word] = []
                self.keyword_index[word].append(entry.id)

        await self.vector_store.upsert(entry)
        return entry

    async def search(
        self,
        query: str,
        top_k: int = 5,
        metadata_filter: dict | None = None,
        min_relevance: float = 0.6,
    ) -> list[MemoryEntry]:
        """н•ҳмқҙлёҢлҰ¬л“ң кІҖмғү: vector similarity + keyword overlap"""

        # 1. Vector search
        query_embedding = await self.embedder.embed(query)
        vector_results = await self.vector_store.search(query, top_k * 2)

        # 2. Keyword search
        keyword_scores = {}
        query_words = [w.lower() for w in query.split() if len(w) > 3]
        for word in query_words:
            if word in self.keyword_index:
                for entry_id in self.keyword_index[word]:
                    keyword_scores[entry_id] = keyword_scores.get(entry_id, 0) + 1

        # 3. н•ҳмқҙлёҢлҰ¬л“ң мҠӨмҪ”м–ҙ кІ°н•©
        results_with_scores = []
        for entry in vector_results:
            if entry.id in keyword_scores:
                keyword_score = keyword_scores[entry.id] / len(query_words)
            else:
                keyword_score = 0.0

            vector_score = np.dot(query_embedding, entry.embedding)  # cosine мң мӮ¬лҸ„ к°Җм •
            hybrid_score = self.alpha * vector_score + (1 - self.alpha) * keyword_score

            # Metadata filter м Ғмҡ©
            if metadata_filter:
                if not all(entry.metadata.get(k) == v for k, v in metadata_filter.items()):
                    continue

            if hybrid_score >= min_relevance:
                results_with_scores.append((entry, hybrid_score))

        # м •л ¬ нӣ„ top_k л°ҳнҷҳ
        results_with_scores.sort(key=lambda x: x[1], reverse=True)
        return [entry for entry, _ in results_with_scores[:top_k]]

from datetime import timedelta


class ForgetPolicy:
    """
    мӢңк°„ кІҪкіјм—җ л”°лҘё importance decayлҘј мһҗлҸҷнҷ”н•ҳлҠ” forget м •мұ….
    м ‘к·јн•ҳм§Җ м•ҠмқҖ кё°м–өмқҖ м җм§„м ҒмңјлЎң importanceлҘј мһғкі  eventually мӮӯм ң.
    """

    def __init__(
        self,
        decay_rate_per_day: float = 0.1,   # л§Өмқј 10% importance к°җмҶҢ
        min_importance: float = 0.1,        # мқҙ мқҙн•ҳмқҙл©ҙ мӮӯм ң лҢҖмғҒ
        check_interval_hours: int = 24,    # 24мӢңк°„л§ҲлӢӨ мІҙнҒ¬
    ):
        self.decay_rate = decay_rate_per_day
        self.min_importance = min_importance

    def compute_decay(self, entry: MemoryEntry) -> float:
        """кІҪкіј мӢңк°„м—җ л”°лҘё decay м җмҲҳ л°ҳнҷҳ"""
        age_days = (datetime.now() - entry.accessed_at).total_seconds() / 86400
        decayed = entry.importance * (1 - self.decay_rate) ** age_days
        return max(decayed, 0.0)

    def should_forget(self, entry: MemoryEntry) -> bool:
        """мӮӯм ң лҢҖмғҒ нҢҗм •"""
        # м ‘к·ј л№ҲлҸ„к°Җ лӮ®кі  мҳӨлһҳлҗң кё°м–ө мҡ°м„  мӮӯм ң
        current_importance = self.compute_decay(entry)
        recency_penalty = 1.0 / (1.0 + age_days * 0.5)
        final_score = current_importance * recency_penalty * (entry.access_count / 10)

        return final_score < self.min_importance

    def gc(self, memory_entries: list[MemoryEntry]) -> list[MemoryEntry]:
        """Forget м •мұ…м—җ л”°лқј entriesлҘј н•„н„°л§Ғ"""
        kept = []
        forgotten = []

        for entry in memory_entries:
            if self.should_forget(entry):
                forgotten.append(entry)
            else:
                kept.append(entry)

        if forgotten:
            print(f"[ForgetPolicy] {len(forgotten)}к°ң н•ӯлӘ© мӮӯм ң. "
                  f"лӮЁмқҖ н•ӯлӘ©: {len(kept)}")

        return kept

мӢңлӮҳлҰ¬мҳӨ 2: мҪ”л”© Agent вҖ” н”„лЎңм қнҠё мғҒнғң м¶”м Ғ

"""

н”„лЎңм қнҠё л””л үнҶ лҰ¬ кө¬мЎ°, л§Ҳм§Җл§ү нҺём§‘ мң„м№ҳ, л№Ңл“ң мғҒнғң л“ұмқ„ м¶”м Ғ.

class CodingAgentMemory:
    """
    мҪ”л”© Agent м „мҡ© л©”лӘЁлҰ¬ вҖ” нҢҢмқј кө¬мЎ°, л№Ңл“ң мғҒнғң, нҳ„мһ¬ мһ‘м—… нҢҢмқј
    """

  3. кІҖмғү н’Ҳм§Ҳмқҙ кі§ мӮ°м¶ңл¬ј н’Ҳм§ҲмқҙлӢӨ: RAGк°Җ к·ёл Үл“Ҝ, Agent Memoryмқҳ кІҖмғү кІ°кіјк°Җ LLMмқҳ reasoning мһ¬лЈҢк°Җ лҗңлӢӨ. "мўӢмқҖ кІҖмғү = мўӢмқҖ лӢөліҖ"мқҙлқјлҠ” мӣҗм№ҷмқҖ л¬ҙм—ҮліҙлӢӨ лЁјм ҖлӢӨ.

мқҙ мІҙнҒ¬лҰ¬мҠӨнҠё н•ҳлӮҳк°Җ н”„лЎңлҚ•м…ҳ Agentмқҳ мӢ лў°м„ұмқ„ н•ң лӢЁкі„ лҒҢм–ҙмҳ¬лҰ°лӢӨ.

  3. Forget м „лһөмқҳ мӢӨм§Ҳм„ұ: "м ҖмһҘн•ҳл©ҙ мўӢмқҖ кІғ"л§Ң к°•мЎ°н•ҳм§Җ м•Ҡкі , л¬ҙн•ң м ҖмһҘмңјлЎң мқён•ң л№„мҡ©/н’Ҳм§Ҳ м Җн•ҳ л¬ём ңлҘј лЁјм Җ м ңкё°н•ҳкі , 4к°Җм§Җ forget м „лһөмқ„ л№„көҗн‘ңлЎң м •лҰ¬. Importance decay мҪ”л“ңлҠ” productionм—җм„ң л°”лЎң мӮ¬мҡ© к°ҖлҠҘ.
